jQuery(function($) {
  // draw the map
  $('#map_canvas').cartis({
    latitude: 42.351505,
    longitude: -71.094455
  });

  // place marker
  var marker = $.cartis.addMapMarker({
    latitude: 42.3515,
    longitude: -71.0944
  });

  // Add hover event handler
  $.cartis.hover(marker, function() {
      $('#caption').show();
    }, function() {
      $('#caption').hide();
    }
  );
});
