jQuery(function($) {
  var gMarker;
  var marker = {
    latitude: 30.291498,
    longitude: -81.390109,
  }

  $('#map_canvas').cartis({
    latitude: 30.291498,
    longitude: -81.390109,
  });

  gMarker = $.cartis.addMapMarker(marker);
  $.cartis.click(gMarker, function() {
    $('#address').toggle();
  });
});
